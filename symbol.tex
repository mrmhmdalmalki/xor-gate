% XOR gate - standard ANSI/IEEE distinctive-shape symbol.
% Like OR but with an extra curved line at the inputs.
\documentclass[border=12pt]{standalone}
\usepackage{circuitikz}
\begin{document}
\begin{circuitikz}[line width=0.8pt, scale=1.4, transform shape]
  \ctikzset{tripoles/thickness=1, bipoles/thickness=1, logic ports/thickness=1}
  \node[xor port] (g) {};
  \draw (g.in 1) -- ++(-1,0) node[left]{$A$};
  \draw (g.in 2) -- ++(-1,0) node[left]{$B$};
  \draw (g.out)  -- ++(1,0)  node[ocirc]{} node[right]{$Y=A\oplus B$};
\end{circuitikz}
\end{document}
